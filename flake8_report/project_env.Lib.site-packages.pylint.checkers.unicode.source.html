141 def _line_length(line: _StrLike, codec: str) -> int:
142 """Get the length of a string like line as displayed in an editor."""
143 if isinstance(line, bytes):
144 decoded = _remove_bom(line, codec).decode(codec, "replace")
145 else:
146 decoded = line
147  
148 stripped = decoded.rstrip("\n")
149  
150 if stripped != decoded:
151 stripped = stripped.rstrip("\r")
152  
153 return len(stripped)

156 def _map_positions_to_result(
157 line: _StrLike,
158 search_dict: dict[_StrLike, _BadChar],
159 new_line: _StrLike,
160 byte_str_length: int = 1,
161 ) -> dict[int, _BadChar]:
162 """Get all occurrences of search dict keys within line.
163  
164 Ignores Windows end of line and can handle bytes as well as string.
165 Also takes care of encodings for which the length of an encoded code point does not
166 default to 8 Bit.
167 """
168  
169 result: dict[int, _BadChar] = {}
170  
171 for search_for, char in search_dict.items():
172 if search_for not in line:
173 continue
174  
175 # Special Handling for Windows '\r\n'
176 if char.unescaped == "\r" and line.endswith(new_line):
177 ignore_pos = len(line) - 2 * byte_str_length
178 else:
179 ignore_pos = None
180  
181 start = 0
182 pos = line.find(search_for, start)
183 while pos > 0:
184 if pos != ignore_pos:
185 # Calculate the column
186 col = int(pos / byte_str_length)
187 result[col] = char
188 start = pos + 1
189 pos = line.find(search_for, start)
190  
191 return result

194 UNICODE_BOMS = {
195 "utf-8": codecs.BOM_UTF8,
196 "utf-16": codecs.BOM_UTF16,
197 "utf-32": codecs.BOM_UTF32,
198 "utf-16le": codecs.BOM_UTF16_LE,
199 "utf-16be": codecs.BOM_UTF16_BE,
200 "utf-32le": codecs.BOM_UTF32_LE,
201 "utf-32be": codecs.BOM_UTF32_BE,
202 }
203 BOM_SORTED_TO_CODEC = OrderedDict(
204 # Sorted by length of BOM of each codec
205 (UNICODE_BOMS[codec], codec)
206 for codec in ("utf-32le", "utf-32be", "utf-8", "utf-16le", "utf-16be")
207 )

209 UTF_NAME_REGEX_COMPILED = re.compile(
210 "utf[ -]?(8|16|32)[ -]?(le|be|)?(sig)?", flags=re.IGNORECASE
211 )

216 return UTF_NAME_REGEX_COMPILED.sub(r"utf-\1\2", codec).lower()

220 """Remove the bom if given from a line."""

222 return encoded

234 def _byte_to_str_length(codec: str) -> int:
235 """Return how many byte are usually(!) a character point."""
236 if codec.startswith("utf-32"):
237 return 4

239 return 2

250 def _fix_utf16_32_line_stream(steam: Iterable[bytes], codec: str) -> Iterable[bytes]:
251 """Handle line ending for UTF16 and UTF32 correctly.
252  
253 Currently, Python simply strips the required zeros after \n after the
254 line ending. Leading to lines that can't be decoded properly
255 """
256 if not codec.startswith("utf-16") and not codec.startswith("utf-32"):
257 yield from steam
258 else:
259 # First we get all the bytes in memory
260 content = b"".join(line for line in steam)
261  
262 new_line = _cached_encode_search("\n", codec)
263  
264 # Now we split the line by the real new line in the correct encoding
265 # we can't use split as it would strip the \n that we need
266 start = 0
267 while True:
268 pos = content.find(new_line, start)
269 if pos >= 0:
  • E203 Whitespace before ':'
270 yield content[start : pos + len(new_line)]
271 else:
272 # Yield the rest and finish
273 if content[start:]:
274 yield content[start:]
275 break
276  
277 start = pos + len(new_line)

280 def extract_codec_from_bom(first_line: bytes) -> str:
281 """Try to extract the codec (unicode only) by checking for the BOM.
282  
283 For details about BOM see https://unicode.org/faq/utf_bom.html#BOM
284  
285 Args:
286 first_line: the first line of a file
287  
288 Returns:
289 a codec name
290  
291 Raises:
292 ValueError: if no codec was found
293 """
294 for bom, codec in BOM_SORTED_TO_CODEC.items():
295 if first_line.startswith(bom):
296 return codec
297  
298 raise ValueError("No BOM found. Could not detect Unicode codec.")

302 """Check characters that could be used to hide bad code to humans.
303  
304 This includes:
305  
306 - Bidirectional Unicode (see https://trojansource.codes/)
307  
308 - Bad ASCII characters (see PEP672)
309  
310 If a programmer requires to use such a character they should use the escaped
311 version, that is also much easier to read and does not depend on the editor used.
312  
313 The Checker also includes a check that UTF-16 and UTF-32 are not used to encode
314 Python files.
315  
316 At the time of writing Python supported only UTF-8. See

318 for background.

383 @classmethod
384 def _find_line_matches(cls, line: bytes, codec: str) -> dict[int, _BadChar]:
385 """Find all matches of BAD_CHARS within line.
386  
387 Args:
388 line: the input
389 codec: that will be used to convert line/or search string into
390  
391 Return:
392 A dictionary with the column offset and the BadASCIIChar
393 """
394 # We try to decode in Unicode to get the correct column offset
395 # if we would use bytes, it could be off because UTF-8 has no fixed length
396 try:
397 line_search = line.decode(codec, errors="strict")
398 search_dict = BAD_ASCII_SEARCH_DICT
399 return _map_positions_to_result(line_search, search_dict, "\n")
400 except UnicodeDecodeError:
401 # If we can't decode properly, we simply use bytes, even so the column offsets
402 # might be wrong a bit, but it is still better then nothing
403 line_search_byte = line
404 search_dict_byte: dict[bytes, _BadChar] = {}
405 for char in BAD_CHARS:
406 # Some characters might not exist in all encodings
407 with contextlib.suppress(UnicodeDecodeError):
408 search_dict_byte[
409 _cached_encode_search(char.unescaped, codec)
410 ] = char
411  
412 return _map_positions_to_result(
413 line_search_byte,
414 search_dict_byte,
415 _cached_encode_search("\n", codec),
416 byte_str_length=_byte_to_str_length(codec),
417 )

419 @staticmethod
420 def _determine_codec(stream: io.BytesIO) -> tuple[str, int]:
421 """Determine the codec from the given stream.
422  
423 first tries https://www.python.org/dev/peps/pep-0263/
424 and if this fails also checks for BOMs of UTF-16 and UTF-32
425 to be future-proof.
426  
427 Args:
428 stream: The byte stream to analyse
429  
430 Returns: A tuple consisting of:
431 - normalized codec name
432 - the line in which the codec was found
433  
434 Raises:
435 SyntaxError: if failing to detect codec
436 """
437 try:
438 # First try to detect encoding with PEP 263
439 # Doesn't work with UTF-16/32 at the time of writing
440 # see https://bugs.python.org/issue1503789
441 codec, lines = detect_encoding(stream.readline)
442  
443 # lines are empty if UTF-8 BOM is found
444 codec_definition_line = len(lines) or 1
445 except SyntaxError as e:
446 # Codec could not be detected by Python, we try manually to check for
447 # UTF 16/32 BOMs, which aren't supported by Python at the time of writing.
448 # This is only included to be future save and handle these codecs as well
449 stream.seek(0)
450 try:
451 codec = extract_codec_from_bom(stream.readline())
452 codec_definition_line = 1
453 except ValueError as ve:
454 # Failed to detect codec, so the syntax error originated not from
455 # UTF16/32 codec usage. So simply raise the error again.
456 raise e from ve
457  
458 return _normalize_codec_name(codec), codec_definition_line

460 def _check_codec(self, codec: str, codec_definition_line: int) -> None:
461 """Check validity of the codec."""
462 if codec != "utf-8":
463 msg = "bad-file-encoding"
464 if self._is_invalid_codec(codec):
465 msg = "invalid-unicode-codec"
466 self.add_message(
467 msg,
468 # Currently Nodes will lead to crashes of pylint
469 # node=node,
470 line=codec_definition_line,
471 end_lineno=codec_definition_line,
472 confidence=pylint.interfaces.HIGH,
473 col_offset=None,
474 end_col_offset=None,
475 )

477 def _check_invalid_chars(self, line: bytes, lineno: int, codec: str) -> None:
478 """Look for chars considered bad."""
479 matches = self._find_line_matches(line, codec)
480 for col, char in matches.items():
481 self.add_message(
482 char.human_code(),
483 # Currently Nodes will lead to crashes of pylint
484 # node=node,
485 line=lineno,
486 end_lineno=lineno,
487 confidence=pylint.interfaces.HIGH,
488 col_offset=col + 1,
489 end_col_offset=col + len(char.unescaped) + 1,
490 )

492 def _check_bidi_chars(self, line: bytes, lineno: int, codec: str) -> None:
493 """Look for Bidirectional Unicode, if we use unicode."""
494 if not self._is_unicode(codec):
495 return
496 for dangerous in BIDI_UNICODE:
497 if _cached_encode_search(dangerous, codec) in line:
498 # Note that we don't add a col_offset on purpose:
499 # Using these unicode characters it depends on the editor
500 # how it displays the location of characters in the line.
501 # So we mark the complete line.
502 self.add_message(
503 "bidirectional-unicode",
504 # Currently Nodes will lead to crashes of pylint
505 # node=node,
506 line=lineno,
507 end_lineno=lineno,
508 # We mark the complete line, as bidi controls make it hard
509 # to determine the correct cursor position within an editor
510 col_offset=0,
511 end_col_offset=_line_length(line, codec),
512 confidence=pylint.interfaces.HIGH,
513 )
514 # We look for bidirectional unicode only once per line
515 # as we mark the complete line anyway
516 break

518 def process_module(self, node: nodes.Module) -> None:
519 """Perform the actual check by checking module stream."""
520 with node.stream() as stream:
521 codec, codec_line = self._determine_codec(stream)
522 self._check_codec(codec, codec_line)
523  
524 stream.seek(0)
525  
526 # Check for invalid content (controls/chars)
527 for (lineno, line) in enumerate(
528 _fix_utf16_32_line_stream(stream, codec), start=1
529 ):
530 if lineno == 1:
531 line = _remove_bom(line, codec)
532 self._check_bidi_chars(line, lineno, codec)
533 self._check_invalid_chars(line, lineno, codec)
