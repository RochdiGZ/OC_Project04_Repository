34 def change_year(current_date: dtdate, year_diff: int) -> dtdate:
35 """
36 Unless the current_date is February 29th, it is fine to just subtract years.
37 If it is a leap day, and we are rolling back to a non-leap year, it will
38 cause a ValueError.
39 Since this is relatively uncommon, just catch the error and roll forward to
40 March 1
41  
42 current_date: date object
43 year_diff: int year delta value, positive or negative
44 """
45 year = current_date.year + year_diff
46 try:
47 return current_date.replace(year=year)
48 except ValueError as e:
49 # ValueError thrown if trying to move date to a non-leap year if the current
50 # date is February 29th
51 if year != 0 and current_date.month == 2 and current_date.day == 29:
52 return current_date.replace(month=3, day=1, year=year)
53 else:
54 raise e
