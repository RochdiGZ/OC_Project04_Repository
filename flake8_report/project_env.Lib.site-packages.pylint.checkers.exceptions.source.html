28 def _builtin_exceptions() -> set[str]:
29 def predicate(obj: Any) -> bool:
30 return isinstance(obj, type) and issubclass(obj, BaseException)
31  
32 members = inspect.getmembers(builtins, predicate)
33 return {exc.__name__ for (_, exc) in members}

36 def _annotated_unpack_infer(
37 stmt: nodes.NodeNG, context: InferenceContext | None = None
38 ) -> Generator[tuple[nodes.NodeNG, SuccessfulInferenceResult], None, None]:
39 """Recursively generate nodes inferred by the given statement.
40  
41 If the inferred value is a list or a tuple, recurse on the elements.
42 Returns an iterator which yields tuples in the format
43 ('original node', 'inferred node').
44 """
45 if isinstance(stmt, (nodes.List, nodes.Tuple)):
46 for elt in stmt.elts:
47 inferred = utils.safe_infer(elt)
48 if inferred and inferred is not astroid.Uninferable:
49 yield elt, inferred
50 return
51 for inferred in stmt.infer(context):
52 if inferred is astroid.Uninferable:
53 continue
54 yield stmt, inferred

57 def _is_raising(body: list[nodes.NodeNG]) -> bool:
58 """Return whether the given statement node raises an exception."""
59 return any(isinstance(node, nodes.Raise) for node in body)

174 def __init__(self, checker: ExceptionsChecker, node: nodes.Raise) -> None:
175 self._checker = checker
176 self._node = node

178 def visit(self, node: SuccessfulInferenceResult) -> None:
179 name = node.__class__.__name__.lower()
180 dispatch_meth = getattr(self, "visit_" + name, None)
181 if dispatch_meth:
182 dispatch_meth(node)
183 else:
184 self.visit_default(node)

286 def _check_misplaced_bare_raise(self, node: nodes.Raise) -> None:
287 # Filter out if it's present in __exit__.
288 scope = node.scope()
289 if (
290 isinstance(scope, nodes.FunctionDef)
291 and scope.is_method()
292 and scope.name == "__exit__"
293 ):
294 return
295  
296 current = node
297 # Stop when a new scope is generated or when the raise
298 # statement is found inside a TryFinally.
299 ignores = (nodes.ExceptHandler, nodes.FunctionDef)
300 while current and not isinstance(current.parent, ignores):
301 current = current.parent
302  
303 expected = (nodes.ExceptHandler,)
304 if not current or not isinstance(current.parent, expected):
305 self.add_message("misplaced-bare-raise", node=node)

307 def _check_bad_exception_cause(self, node: nodes.Raise) -> None:
308 """Verify that the exception cause is properly set.
309  
310 An exception cause can be only `None` or an exception.
311 """
312 cause = utils.safe_infer(node.cause)
313 if cause in (astroid.Uninferable, None):
314 return
315  
316 if isinstance(cause, nodes.Const):
317 if cause.value is not None:
318 self.add_message("bad-exception-cause", node=node, confidence=INFERENCE)
319 elif not isinstance(cause, nodes.ClassDef) and not utils.inherit_from_std_ex(
320 cause
321 ):
322 self.add_message("bad-exception-cause", node=node, confidence=INFERENCE)

324 def _check_raise_missing_from(self, node: nodes.Raise) -> None:
325 if node.exc is None:
326 # This is a plain `raise`, raising the previously-caught exception. No need for a
327 # cause.
328 return
329 # We'd like to check whether we're inside an `except` clause:
330 containing_except_node = utils.find_except_wrapper_node_in_scope(node)
331 if not containing_except_node:
332 return
333 # We found a surrounding `except`! We're almost done proving there's a
334 # `raise-missing-from` here. The only thing we need to protect against is that maybe
335 # the `raise` is raising the exception that was caught, possibly with some shenanigans
336 # like `exc.with_traceback(whatever)`. We won't analyze these, we'll just assume
337 # there's a violation on two simple cases: `raise SomeException(whatever)` and `raise
338 # SomeException`.
339 if containing_except_node.name is None:
340 # The `except` doesn't have an `as exception:` part, meaning there's no way that
341 # the `raise` is raising the same exception.
342 class_of_old_error = "Exception"
343 if isinstance(containing_except_node.type, (nodes.Name, nodes.Tuple)):
344 # 'except ZeroDivisionError' or 'except (ZeroDivisionError, ValueError)'
345 class_of_old_error = containing_except_node.type.as_string()
346 self.add_message(
347 "raise-missing-from",
348 node=node,
349 args=(
350 f"'except {class_of_old_error} as exc' and ",
351 node.as_string(),
352 "exc",
353 ),
354 confidence=HIGH,
355 )
356 elif (
357 isinstance(node.exc, nodes.Call)
358 and isinstance(node.exc.func, nodes.Name)
359 or isinstance(node.exc, nodes.Name)
360 and node.exc.name != containing_except_node.name.name
361 ):
362 # We have a `raise SomeException(whatever)` or a `raise SomeException`
363 self.add_message(
364 "raise-missing-from",
365 node=node,
366 args=("", node.as_string(), containing_except_node.name.name),
367 confidence=HIGH,
368 )

370 def _check_catching_non_exception(
371 self,
372 handler: nodes.ExceptHandler,
373 exc: SuccessfulInferenceResult,
374 part: nodes.NodeNG,
375 ) -> None:
376 if isinstance(exc, nodes.Tuple):
377 # Check if it is a tuple of exceptions.
378 inferred = [utils.safe_infer(elt) for elt in exc.elts]
379 if any(node is astroid.Uninferable for node in inferred):
380 # Don't emit if we don't know every component.
381 return
382 if all(
383 node
384 and (utils.inherit_from_std_ex(node) or not utils.has_known_bases(node))
385 for node in inferred
386 ):
387 return
388  
389 if not isinstance(exc, nodes.ClassDef):
390 # Don't emit the warning if the inferred stmt
391 # is None, but the exception handler is something else,
392 # maybe it was redefined.
393 if isinstance(exc, nodes.Const) and exc.value is None:
394 if (
395 isinstance(handler.type, nodes.Const) and handler.type.value is None
396 ) or handler.type.parent_of(exc):
397 # If the exception handler catches None or
398 # the exception component, which is None, is
399 # defined by the entire exception handler, then
400 # emit a warning.
401 self.add_message(
402 "catching-non-exception",
403 node=handler.type,
404 args=(part.as_string(),),
405 )
406 else:
407 self.add_message(
408 "catching-non-exception",
409 node=handler.type,
410 args=(part.as_string(),),
411 )
412 return
413  
414 if (
415 not utils.inherit_from_std_ex(exc)
416 and exc.name not in self._builtin_exceptions
417 ):
418 if utils.has_known_bases(exc):
419 self.add_message(
420 "catching-non-exception", node=handler.type, args=(exc.name,)
421 )

423 def _check_try_except_raise(self, node: nodes.TryExcept) -> None:
424 def gather_exceptions_from_handler(
425 handler: nodes.ExceptHandler,
426 ) -> list[InferenceResult] | None:
427 exceptions: list[InferenceResult] = []
428 if handler.type:
429 exceptions_in_handler = utils.safe_infer(handler.type)
430 if isinstance(exceptions_in_handler, nodes.Tuple):
431 exceptions = list(
432 {
433 exception
434 for exception in exceptions_in_handler.elts
435 if isinstance(exception, nodes.Name)
436 }
437 )
438 elif exceptions_in_handler:
439 exceptions = [exceptions_in_handler]
440 else:
441 # Break when we cannot infer anything reliably.
442 return None
443 return exceptions
444  
445 bare_raise = False
446 handler_having_bare_raise = None
447 exceptions_in_bare_handler: list[InferenceResult] | None = []
448 for handler in node.handlers:
449 if bare_raise:
450 # check that subsequent handler is not parent of handler which had bare raise.
451 # since utils.safe_infer can fail for bare except, check it before.
452 # also break early if bare except is followed by bare except.
453  
454 excs_in_current_handler = gather_exceptions_from_handler(handler)
455 if not excs_in_current_handler:
456 break
457 if exceptions_in_bare_handler is None:
458 # It can be `None` when the inference failed
459 break
460 for exc_in_current_handler in excs_in_current_handler:
461 inferred_current = utils.safe_infer(exc_in_current_handler)
462 if any(
463 utils.is_subclass_of(utils.safe_infer(e), inferred_current)
464 for e in exceptions_in_bare_handler
465 ):
466 bare_raise = False
467 break
468  
469 # `raise` as the first operator inside the except handler
470 if _is_raising([handler.body[0]]):
471 # flags when there is a bare raise
472 if handler.body[0].exc is None:
473 bare_raise = True
474 handler_having_bare_raise = handler
475 exceptions_in_bare_handler = gather_exceptions_from_handler(handler)
476 else:
477 if bare_raise:
478 self.add_message("try-except-raise", node=handler_having_bare_raise)

494 @utils.only_required_for_messages(
495 "bare-except",
496 "broad-except",
497 "try-except-raise",
498 "binary-op-exception",
499 "bad-except-order",
500 "catching-non-exception",
501 "duplicate-except",
502 )
503 def visit_tryexcept(self, node: nodes.TryExcept) -> None:
504 """Check for empty except."""
505 self._check_try_except_raise(node)
506 exceptions_classes: list[Any] = []
507 nb_handlers = len(node.handlers)
508 for index, handler in enumerate(node.handlers):
509 if handler.type is None:
510 if not _is_raising(handler.body):
511 self.add_message("bare-except", node=handler)
512  
513 # check if an "except:" is followed by some other
514 # except
515 if index < (nb_handlers - 1):
516 msg = "empty except clause should always appear last"
517 self.add_message("bad-except-order", node=node, args=msg)
518  
519 elif isinstance(handler.type, nodes.BoolOp):
520 self.add_message(
521 "binary-op-exception", node=handler, args=handler.type.op
522 )
523 else:
524 try:
525 exceptions = list(_annotated_unpack_infer(handler.type))
526 except astroid.InferenceError:
527 continue
528  
529 for part, exception in exceptions:
530 if isinstance(
531 exception, astroid.Instance
532 ) and utils.inherit_from_std_ex(exception):
533 exception = exception._proxied
534  
535 self._check_catching_non_exception(handler, exception, part)
536  
537 if not isinstance(exception, nodes.ClassDef):
538 continue
539  
540 exc_ancestors = [
541 anc
542 for anc in exception.ancestors()
543 if isinstance(anc, nodes.ClassDef)
544 ]
545  
546 for previous_exc in exceptions_classes:
547 if previous_exc in exc_ancestors:
548 msg = f"{previous_exc.name} is an ancestor class of {exception.name}"
549 self.add_message(
550 "bad-except-order", node=handler.type, args=msg
551 )
552 if (
553 exception.name in self.linter.config.overgeneral_exceptions
554 and exception.root().name == utils.EXCEPTIONS_MODULE
555 and not _is_raising(handler.body)
556 ):
557 self.add_message(
558 "broad-except", args=exception.name, node=handler.type
559 )
560  
561 if exception in exceptions_classes:
562 self.add_message(
563 "duplicate-except", args=exception.name, node=handler.type
564 )
565  
566 exceptions_classes += [exc for _, exc in exceptions]
