106 def looks_like_typing_typevar_or_newtype(node):
107 func = node.func
108 if isinstance(func, Attribute):
109 return func.attrname in TYPING_TYPEVARS
110 if isinstance(func, Name):
111 return func.name in TYPING_TYPEVARS
112 return False

115 def infer_typing_typevar_or_newtype(node, context_itton=None):
116 """Infer a typing.TypeVar(...) or typing.NewType(...) call"""
117 try:
118 func = next(node.func.infer(context=context_itton))
119 except (InferenceError, StopIteration) as exc:
120 raise UseInferenceDefault from exc
121  
122 if func.qname() not in TYPING_TYPEVARS_QUALIFIED:
123 raise UseInferenceDefault
124 if not node.args:
125 raise UseInferenceDefault
126 # Cannot infer from a dynamic class name (f-string)
127 if isinstance(node.args[0], JoinedStr):
128 raise UseInferenceDefault
129  
130 typename = node.args[0].as_string().strip("'")
131 node = extract_node(TYPING_TYPE_TEMPLATE.format(typename))
132 return node.infer(context=context_itton)

135 def _looks_like_typing_subscript(node):
136 """Try to figure out if a Subscript node *might* be a typing-related subscript"""
137 if isinstance(node, Name):
138 return node.name in TYPING_MEMBERS
139 if isinstance(node, Attribute):
140 return node.attrname in TYPING_MEMBERS
141 if isinstance(node, Subscript):
142 return _looks_like_typing_subscript(node.value)
143 return False

146 def infer_typing_attr(
147 node: Subscript, ctx: context.InferenceContext | None = None
148 ) -> Iterator[ClassDef]:
149 """Infer a typing.X[...] subscript"""
150 try:
151 value = next(node.value.infer()) # type: ignore[union-attr] # value shouldn't be None for Subscript.
152 except (InferenceError, StopIteration) as exc:
153 raise UseInferenceDefault from exc
154  
155 if not value.qname().startswith("typing.") or value.qname() in TYPING_ALIAS:
156 # If typing subscript belongs to an alias handle it separately.
157 raise UseInferenceDefault
158  
159 if isinstance(value, ClassDef) and value.qname() in {
160 "typing.Generic",
161 "typing.Annotated",
162 "typing_extensions.Annotated",
163 }:
164 # typing.Generic and typing.Annotated (PY39) are subscriptable
165 # through __class_getitem__. Since astroid can't easily
166 # infer the native methods, replace them for an easy inference tip
167 func_to_add = _extract_single_node(CLASS_GETITEM_TEMPLATE)
168 value.locals["__class_getitem__"] = [func_to_add]
169 if (
170 isinstance(node.parent, ClassDef)
171 and node in node.parent.bases
172 and getattr(node.parent, "__cache", None)
173 ):
174 # node.parent.slots is evaluated and cached before the inference tip
175 # is first applied. Remove the last result to allow a recalculation of slots
176 cache = node.parent.__cache # type: ignore[attr-defined] # Unrecognized getattr
177 if cache.get(node.parent.slots) is not None:
178 del cache[node.parent.slots]
179 return iter([value])
180  
181 node = extract_node(TYPING_TYPE_TEMPLATE.format(value.qname().split(".")[-1]))
182 return node.infer(context=ctx)

200 def infer_typedDict( # pylint: disable=invalid-name
201 node: FunctionDef, ctx: context.InferenceContext | None = None
202 ) -> Iterator[ClassDef]:
203 """Replace TypedDict FunctionDef with ClassDef."""
204 class_def = ClassDef(
205 name="TypedDict",
206 lineno=node.lineno,
207 col_offset=node.col_offset,
208 parent=node.parent,
209 )
210 class_def.postinit(bases=[extract_node("dict")], body=[], decorators=None)
211 func_to_add = _extract_single_node("dict")
212 class_def.locals["__call__"] = [func_to_add]
213 return iter([class_def])

260 def infer_typing_alias(
261 node: Call, ctx: context.InferenceContext | None = None
262 ) -> Iterator[ClassDef]:
263 """
264 Infers the call to _alias function
265 Insert ClassDef, with same name as aliased class,
266 in mro to simulate _GenericAlias.
267  
268 :param node: call node
269 :param context: inference context
270 """
271 if (
272 not isinstance(node.parent, Assign)
273 or not len(node.parent.targets) == 1
274 or not isinstance(node.parent.targets[0], AssignName)
275 ):
276 raise UseInferenceDefault
277 try:
278 res = next(node.args[0].infer(context=ctx))
279 except StopIteration as e:
280 raise InferenceError(node=node.args[0], context=context) from e
281  
282 assign_name = node.parent.targets[0]
283  
284 class_def = ClassDef(
285 name=assign_name.name,
286 lineno=assign_name.lineno,
287 col_offset=assign_name.col_offset,
288 parent=node.parent,
289 )
290 if res != Uninferable and isinstance(res, ClassDef):
291 # Only add `res` as base if it's a `ClassDef`
292 # This isn't the case for `typing.Pattern` and `typing.Match`
293 class_def.postinit(bases=[res], body=[], decorators=None)
294  
295 maybe_type_var = node.args[1]
296 if (
297 not PY39_PLUS
298 and not (isinstance(maybe_type_var, Tuple) and not maybe_type_var.elts)
299 or PY39_PLUS
300 and isinstance(maybe_type_var, Const)
301 and maybe_type_var.value > 0
302 ):
303 # If typing alias is subscriptable, add `__class_getitem__` to ClassDef
304 func_to_add = _extract_single_node(CLASS_GETITEM_TEMPLATE)
305 class_def.locals["__class_getitem__"] = [func_to_add]
306 else:
307 # If not, make sure that `__class_getitem__` access is forbidden.
308 # This is an issue in cases where the aliased class implements it,
309 # but the typing alias isn't subscriptable. E.g., `typing.ByteString` for PY39+
310 _forbid_class_getitem_access(class_def)
311 return iter([class_def])

314 def _looks_like_special_alias(node: Call) -> bool:
315 """Return True if call is for Tuple or Callable alias.
316  
317 In PY37 and PY38 the call is to '_VariadicGenericAlias' with 'tuple' as
318 first argument. In PY39+ it is replaced by a call to '_TupleType'.
319  
320 PY37: Tuple = _VariadicGenericAlias(tuple, (), inst=False, special=True)
321 PY39: Tuple = _TupleType(tuple, -1, inst=False, name='Tuple')
322  
323  
324 PY37: Callable = _VariadicGenericAlias(collections.abc.Callable, (), special=True)
325 PY39: Callable = _CallableType(collections.abc.Callable, 2)
326 """
327 return isinstance(node.func, Name) and (
328 not PY39_PLUS
329 and node.func.name == "_VariadicGenericAlias"
330 and (
331 isinstance(node.args[0], Name)
332 and node.args[0].name == "tuple"
333 or isinstance(node.args[0], Attribute)
334 and node.args[0].as_string() == "collections.abc.Callable"
335 )
336 or PY39_PLUS
337 and (
338 node.func.name == "_TupleType"
339 and isinstance(node.args[0], Name)
340 and node.args[0].name == "tuple"
341 or node.func.name == "_CallableType"
342 and isinstance(node.args[0], Attribute)
343 and node.args[0].as_string() == "collections.abc.Callable"
344 )
345 )

348 def infer_special_alias(
349 node: Call, ctx: context.InferenceContext | None = None
350 ) -> Iterator[ClassDef]:
351 """Infer call to tuple alias as new subscriptable class typing.Tuple."""
352 if not (
353 isinstance(node.parent, Assign)
354 and len(node.parent.targets) == 1
355 and isinstance(node.parent.targets[0], AssignName)
356 ):
357 raise UseInferenceDefault
358 try:
359 res = next(node.args[0].infer(context=ctx))
360 except StopIteration as e:
361 raise InferenceError(node=node.args[0], context=context) from e
362  
363 assign_name = node.parent.targets[0]
364 class_def = ClassDef(
365 name=assign_name.name,
366 parent=node.parent,
367 )
368 class_def.postinit(bases=[res], body=[], decorators=None)
369 func_to_add = _extract_single_node(CLASS_GETITEM_TEMPLATE)
370 class_def.locals["__class_getitem__"] = [func_to_add]
371 return iter([class_def])

383 def infer_typing_cast(
384 node: Call, ctx: context.InferenceContext | None = None
385 ) -> Iterator[NodeNG]:
386 """Infer call to cast() returning same type as casted-from var"""
387 if not isinstance(node.func, (Name, Attribute)):
388 raise UseInferenceDefault
389  
390 try:
391 func = next(node.func.infer(context=ctx))
392 except (InferenceError, StopIteration) as exc:
393 raise UseInferenceDefault from exc
394 if (
395 not isinstance(func, FunctionDef)
396 or func.qname() != "typing.cast"
397 or len(node.args) != 2
398 ):
399 raise UseInferenceDefault
400  
401 return node.args[1].infer(context=ctx)
