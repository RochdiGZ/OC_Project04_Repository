5 """A similarities / code duplication command line tool and pylint checker.
6  
7 The algorithm is based on comparing the hash value of n successive lines of a file.
8 First the files are read and any line that doesn't fulfill requirement are removed (comments, docstrings...)
9 Those stripped lines are stored in the LineSet class which gives access to them.
  • E501 Line too long (152 > 119 characters)
10 Then each index of the stripped lines collection is associated with the hash of n successive entries of the stripped lines starting at the current index
11 (n is the minimum common lines option).
  • E501 Line too long (148 > 119 characters)
12 The common hashes between both linesets are then looked for. If there are matches, then the match indices in both linesets are stored and associated
13 with the corresponding couples (start line number/end line number) in both files.
  • E501 Line too long (150 > 119 characters)
14 This association is then post-processed to handle the case of successive matches. For example if the minimum common lines setting is set to four, then
  • E501 Line too long (150 > 119 characters)
15 the hashes are computed with four lines. If one of match indices couple (12, 34) is the successor of another one (11, 33) then it means that there are
16 in fact five lines which are common.
  • E501 Line too long (141 > 119 characters)
17 Once post-processed the values of association table are the result looked for, i.e start and end lines numbers of common lines in both files.
18 """

209 def hash_lineset(
210 lineset: LineSet, min_common_lines: int = DEFAULT_MIN_SIMILARITY_LINE
211 ) -> tuple[HashToIndex_T, IndexToLines_T]:
212 """Return two dicts.
213  
214 The first associates the hash of successive stripped lines of a lineset
215 to the indices of the starting lines.
216 The second dict, associates the index of the starting line in the lineset's stripped lines to the
217 couple [start, end] lines number in the corresponding file.
218  
219 :param lineset: lineset object (i.e the lines in a file)
220 :param min_common_lines: number of successive lines that are used to compute the hash
221 :return: a dict linking hashes to corresponding start index and a dict that links this
222 index to the start and end lines in the file
223 """
224 hash2index = defaultdict(list)
225 index2lines = {}
226 # Comments, docstring and other specific patterns maybe excluded -> call to stripped_lines
227 # to get only what is desired
228 lines = tuple(x.text for x in lineset.stripped_lines)
229 # Need different iterators on same lines but each one is shifted 1 from the precedent
230 shifted_lines = [iter(lines[i:]) for i in range(min_common_lines)]
231  
232 for i, *succ_lines in enumerate(zip(*shifted_lines)):
233 start_linenumber = LineNumber(lineset.stripped_lines[i].line_number)
234 try:
235 end_linenumber = lineset.stripped_lines[i + min_common_lines].line_number
236 except IndexError:
237 end_linenumber = LineNumber(lineset.stripped_lines[-1].line_number + 1)
238  
239 index = Index(i)
240 index2lines[index] = SuccessiveLinesLimits(
241 start=start_linenumber, end=end_linenumber
242 )
243  
244 l_c = LinesChunk(lineset.name, index, *succ_lines)
245 hash2index[l_c].append(index)
246  
247 return hash2index, index2lines

250 def remove_successive(all_couples: CplIndexToCplLines_T) -> None:
251 """Removes all successive entries in the dictionary in argument.
252  
253 :param all_couples: collection that has to be cleaned up from successive entries.
254 The keys are couples of indices that mark the beginning of common entries
255 in both linesets. The values have two parts. The first one is the couple
256 of starting and ending line numbers of common successive lines in the first file.
257 The second part is the same for the second file.
258  
259 For example consider the following dict:
260  
261 >>> all_couples
262 {(11, 34): ([5, 9], [27, 31]),
263 (23, 79): ([15, 19], [45, 49]),
264 (12, 35): ([6, 10], [28, 32])}
265  
266 There are two successive keys (11, 34) and (12, 35).
267 It means there are two consecutive similar chunks of lines in both files.
268 Thus remove last entry and update the last line numbers in the first entry
269  
270 >>> remove_successive(all_couples)
271 >>> all_couples
272 {(11, 34): ([5, 10], [27, 32]),
273 (23, 79): ([15, 19], [45, 49])}
274 """
275 couple: LineSetStartCouple
276 for couple in tuple(all_couples.keys()):
277 to_remove = []
278 test = couple.increment(Index(1))
279 while test in all_couples:
280 all_couples[couple].first_file.end = all_couples[test].first_file.end
281 all_couples[couple].second_file.end = all_couples[test].second_file.end
282 all_couples[couple].effective_cmn_lines_nb += 1
283 to_remove.append(test)
284 test = test.increment(Index(1))
285  
286 for target in to_remove:
287 try:
288 all_couples.pop(target)
289 except KeyError:
290 pass

463 def _find_common(
464 self, lineset1: LineSet, lineset2: LineSet
465 ) -> Generator[Commonality, None, None]:
466 """Find similarities in the two given linesets.
467  
468 This the core of the algorithm.
  • E501 Line too long (126 > 119 characters)
469 The idea is to compute the hashes of a minimal number of successive lines of each lineset and then compare the hashes.
  • E501 Line too long (120 > 119 characters)
470 Every match of such comparison is stored in a dict that links the couple of starting indices in both linesets to
471 the couple of corresponding starting and ending lines in both files.
  • E501 Line too long (129 > 119 characters)
472 Last regroups all successive couples in a bigger one. It allows to take into account common chunk of lines that have more
473 than the minimal number of successive lines required.

480 lineset1, self.namespace.min_similarity_lines

483 lineset2, self.namespace.min_similarity_lines

490 hash_1 & hash_2, key=lambda m: hash_to_index_1[m][0]

493 # all_couples is a dict that links the couple of indices in both linesets that mark the beginning of
494 # successive common lines, to the corresponding starting and ending number lines in both files
495 all_couples: CplIndexToCplLines_T = {}

648 lineno, current_line_is_import
649 )
650 if current_line_is_import:
651 line = ""

654 if ignore_signatures and lineno in signature_lines:
655 line = ""
656 if line:
657 strippedlines.append(
658 LineSpecifs(text=line, line_number=LineNumber(lineno - 1))
659 )
660 return strippedlines

841 def close(self) -> None:
842 """Compute and display similarities on closing (i.e. end of parsing)."""
843 total = sum(len(lineset) for lineset in self.linesets)
844 duplicated = 0
845 stats = self.linter.stats
846 for num, couples in self._compute_sims():
847 msg = []
848 lineset = start_line = end_line = None
849 for lineset, start_line, end_line in couples:
850 msg.append(f"=={lineset.name}:[{start_line}:{end_line}]")
851 msg.sort()
852  
853 if lineset:
854 for line in lineset.real_lines[start_line:end_line]:
855 msg.append(line.rstrip())
856  
857 self.add_message("R0801", args=(len(couples), "\n".join(msg)))
858 duplicated += num * (len(couples) - 1)
859 stats.nb_duplicated_lines += int(duplicated)
860 stats.percent_duplicated_lines += float(total and duplicated * 100.0 / total)
