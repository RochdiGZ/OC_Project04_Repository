18 class ConfusingConsecutiveElifChecker(BaseChecker):
19 """Checks if "elif" is used right after an indented block that finishes with "if" or
20 "elif" itself.
21 """
22  
23 name = "confusing_elif"
24 msgs = {
25 "R5601": (
  • E501 Line too long (121 > 119 characters)
26 "Consecutive elif with differing indentation level, consider creating a function to separate the inner elif",
27 "confusing-consecutive-elif",
28 "Used when an elif statement follows right after an indented block which itself ends with if or elif. "
29 "It may not be ovious if the elif statement was willingly or mistakenly unindented. "
30 "Extracting the indented if statement into a separate function might avoid confusion and prevent errors.",
31 )
32 }

34 @only_required_for_messages("confusing-consecutive-elif")
35 def visit_if(self, node: nodes.If) -> None:
36 body_ends_with_if = isinstance(
37 node.body[-1], nodes.If
38 ) and self._has_no_else_clause(node.body[-1])
39 if node.has_elif_block() and body_ends_with_if:
40 self.add_message("confusing-consecutive-elif", node=node.orelse[0])
41  
42 @staticmethod
43 def _has_no_else_clause(node: nodes.If) -> bool:
44 orelse = node.orelse
45 while orelse and isinstance(orelse[0], nodes.If):
46 orelse = orelse[0].orelse
47 if not orelse or isinstance(orelse[0], nodes.If):
48 return True
49 return False
