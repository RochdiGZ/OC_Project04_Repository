30 def _open_graph(self) -> None:
31 """Emit the header lines."""
32 self.emit("@startuml " + self.title)
33 if not self.use_automatic_namespace:
34 self.emit("set namespaceSeparator none")
35 if self.layout:
36 if self.layout is Layout.LEFT_TO_RIGHT:
37 self.emit("left to right direction")
38 elif self.layout is Layout.TOP_TO_BOTTOM:
39 self.emit("top to bottom direction")
40 else:
41 raise ValueError(
  • E501 Line too long (126 > 119 characters)
42 f"Unsupported layout {self.layout}. PlantUmlPrinter only supports left to right and top to bottom layout."

45 def emit_node(
46 self,
47 name: str,
48 type_: NodeType,
49 properties: NodeProperties | None = None,
50 ) -> None:
51 """Create a new node.
52  
53 Nodes can be classes, packages, participants etc.
54 """
55 if properties is None:
56 properties = NodeProperties(label=name)
57 stereotype = " << interface >>" if type_ is NodeType.INTERFACE else ""
58 nodetype = self.NODES[type_]
59 if properties.color and properties.color != self.DEFAULT_COLOR:
60 color = f" #{properties.color}"
61 else:
62 color = ""
63 body = []
64 if properties.attrs:
65 body.extend(properties.attrs)
66 if properties.methods:
67 for func in properties.methods:
68 args = self._get_method_arguments(func)
69 line = f"{func.name}({', '.join(args)})"
70 if func.returns:
71 line += " -> " + get_annotation_label(func.returns)
72 body.append(line)
73 label = properties.label if properties.label is not None else name
74 if properties.fontcolor and properties.fontcolor != self.DEFAULT_COLOR:
75 label = f"<color:{properties.fontcolor}>{label}</color>"
76 self.emit(f'{nodetype} "{label}" as {name}{stereotype}{color} {{')
77 self._inc_indent()
78 for line in body:
79 self.emit(line)
80 self._dec_indent()
81 self.emit("}")

83 def emit_edge(
84 self,
85 from_node: str,
86 to_node: str,
87 type_: EdgeType,
88 label: str | None = None,
89 ) -> None:
90 """Create an edge from one node to another to display relationships."""
91 edge = f"{from_node} {self.ARROWS[type_]} {to_node}"
92 if label:
93 edge += f" : {label}"
94 self.emit(edge)
